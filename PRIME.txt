DECLARE
N NUMBER := &N;
I NUMBER;
C NUMBER :=0;
BEGIN
IF N<2 THEN
DBMS_OUTPUT.PUT_LINE('THE NUMBER' ||N|| ' IS NOT PRIME');
ELSE
FOR I IN 2 .. N - 1 LOOP
IF MOD (N,I)=0 THEN
C := 1;
EXIT;
END IF;
END LOOP;
IF C = 0 THEN 
DBMS_OUTPUT.PUT_LINE('THE NUMBER' ||N|| ' IS PRIME');
ELSE
DBMS_OUTPUT.PUT_LINE('THE NUMBER' ||N|| ' IS NOT PRIME');
END IF;
END IF;
END;
/
